@init: ->
    events = @expresser.events
    logger = @expresser.logger
    settings = @expresser.settings

    logger.debug "AWS.init"
    events.emit "AWS.before.init"

    # Init the implemented AWS modules.
    @dynamodb.init this
    @s3.init this
    @sns.init this

    @setEvents()

    events.emit "AWS.on.init"
    delete @init
