Class: Cron

Defined in: plugins/cron/index.coffee

Overview

class

Examples:

Sample job object, alerts user every minute (60 seconds). 

var myJob = {
  id: "alertJob",
  schedule: 60,
  callback: function(job) { alertUser(mydata) }
}

Sample job object, sends email every day at 10AM and 5PM. 

var otherJob = {
  id: "my mail job",
  schedule: ["10:00:00", "17:00:00"],
  callback: function(job) { mail.send(something) }
}
